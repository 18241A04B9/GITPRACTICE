```groovy
pipeline {
    agent none   // No global agent, each stage defines its own

    stages {
        stage('NodeJS Stage') {
            agent {
                docker {
                    image 'node:18'   // You can change version if needed
                    args '-v /tmp:/tmp'  // optional: mount tmp for caching
                }
            }
            steps {
                echo 'Welcome to Jenkins Sanjeev (from NodeJS container)'
            }
        }

        stage('Maven Stage') {
            agent {
                docker {
                    image 'maven:3.9.6-eclipse-temurin-17'
                    args '-v /root/.m2:/root/.m2'  // optional: cache maven repo
                }
            }
            steps {
                echo 'Welcome to Jenkins Sanjeev (from Maven container)'
            }
        }
    }
}
```
